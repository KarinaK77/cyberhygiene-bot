    def check_answer(self, user_id, example_id, user_answer):

        db.save_phishing_attempt(
            user_id=user_id,
            example_id=example_id,
            user_answer=user_answer,
            is_correct=is_correct,
            correct_type=example['type']
        )
        
        # –ù–∞—á–∏—Å–ª—è–µ–º –±–∞–ª–ª—ã –∑–∞ –ø—Ä–∞–≤–∏–ª—å–Ω—ã–π –æ—Ç–≤–µ—Ç
